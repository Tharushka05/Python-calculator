# Python-calculator
#This for Python calculator just 12 lines of code

num1 = float(input("Enter a number;"))
operation = input("Enter an operation;")
num2 = float(input("Enter another number;"))

if operation == "+":
    print(num1+num2)
if operation == "-":
    print(num1-num2)
if operation == "*":
    print(num1*num2)
if operation == "/":
    print(num1/num2)
